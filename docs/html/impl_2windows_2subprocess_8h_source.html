242  static void append_argument(const std::string &arg, std::string &commandline) {
243  if (!arg.empty() && arg.find_first_of(" \t\n\v\"") == arg.npos) {
244  commandline.append(arg);
245  } else {
246  // quote
247  commandline.push_back('"');
248  for (auto it = arg.begin(); it != arg.end(); ++it) {
249  unsigned num_slashes = 0;
250 
251  while (it != arg.end() && *it == '\\') {
252  ++it;
253  ++num_slashes;
254  }
255 
256  if (it == arg.end()) {
257  //
258  // Escape all backslashes, but let the terminating
259  // double quotation mark we add below be interpreted
260  // as a metacharacter.
261  //
262  commandline.append(num_slashes * 2, '\\');
263  break;
264  } else if (*it == '"') {
265  //
266  // Escape all backslashes and the following
267  // double quotation mark.
268  //
269  commandline.append(num_slashes * 2 + 1, '\\');
270  commandline.push_back(*it);
271  } else {
272  //
273  // Backslashes aren't special here.
274  //
275  commandline.append(num_slashes, '\\');
